\newcommand{\ppOptPriceFormula}{%
\ppOptPrice = \ppCost + \plannedSoftIncome
}

\ppOptPriceFormula,
\end{equation}
\begin{explanation}
где & $ \ppCost $ & себестоимость программы, \ye; \\
    & $ \plannedSoftIncome $ & плановая прибыль на программу, \ye
\end{explanation}
}
